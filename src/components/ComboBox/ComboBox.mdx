## Example

```jsx
import { ComboBox } from 'ventileco-ui'

const comboBoxList = [
  { value: 'apple', label: 'Apple', disabled: true },
  { value: 'kiwi', label: 'Kiwi' },
  { value: 'peach', label: 'Peach', disabled: true },
  { value: 'grape', label: 'Grape' },
  { value: 'aloe', label: 'Aloe' },
  { value: 'apple2', label: 'Apple2' },
  { value: 'banana2', label: 'Banana2', disabled: true },
  { value: 'kiwi2', label: 'Kiwi2' },
  { value: 'peach2', label: 'Peach2', disabled: true },
  { value: 'grape2', label: 'Grape2' },
  { value: 'aloe2', label: 'Aloe2' },
]

function Example() {
 const [value, setValue] = useState<string>()

  const animationProps = {
    variants: {
      hidden: { opacity: 0, y: 20 },
      visible: { opacity: 1, y: 0 },
    },
    initial: 'hidden',
    animate: 'visible',
    viewport: { once: true },
    transition: { duration: 0.3 },
  }

  return (
     <ComboBox
      value={value}
      setValue={setValue}
      list={comboBoxList}
      className="w-[240px] text-sm"
    >
      <ComboBox.Trigger>
        <div className="flex w-full items-center rounded-md border border-black px-3 py-2">
          <ComboBox.Input placeholder="ğŸ Fruits" />
          <ComboBox.ClearButton>
            <svg
              xmlns="http://www.w3.org/2000/svg"
              fill="none"
              viewBox="0 0 24 24"
              strokeWidth="1.5"
              stroke="currentColor"
              className="size-4"
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                d="M6 18 18 6M6 6l12 12"
              />
            </svg>
          </ComboBox.ClearButton>
        </div>
      </ComboBox.Trigger>
      <ComboBox.List
        as={motion.ul}
        {...animationProps}
        className='bg-white" absolute z-[200] max-h-[240px] w-full overflow-hidden overflow-y-scroll rounded-md border'
      >
        {({ optionList }) =>
          optionList.length !== 0 ? (
            <>
              {optionList.map((item) => (
                <ComboBox.Item key={item.value} item={item}>
                  {({ isSelected, isDisabled, isFocused }) => (
                    <div
                      className={`${isSelected && 'text-blue-600'} ${isDisabled && 'text-gray-300'} ${isFocused && 'bg-blue-100'} w-full px-3 py-2 hover:bg-gray-100`}
                    >
                      {item.label}
                    </div>
                  )}
                </ComboBox.Item>
              ))}
            </>
          ) : (
            <div className="w-full rounded-md border bg-white px-3 py-2 text-black/50">
              No Results
            </div>
          )
        }
      </ComboBox.List>
    </ComboBox>
  )
}
```
