# Accordion

사용자가 섹션을 열거나 닫을 수 있는 접히고 펼쳐지는 인터페이스 요소

## Example

```jsx
import { Accordion } from 'ventileco-ui'

const accordionList = [
  { id: 0, title: 'Title1', content: 'Content1' },
  { id: 1, title: 'Title2', content: 'Content2' },
  { id: 2, title: 'Title3', content: 'Content3' },
]

export function Example() {
  const animationProps = {
    variants: {
      hidden: { opacity: 0, y: 20 },
      visible: { opacity: 1, y: 0 },
    },
    initial: 'hidden',
    animate: 'visible',
    viewport: { once: true },
    transition: { duration: 0.3 },
  }

  return (
    <div className="w-[240px]">
      <Accordion
        multiple
        className="flex flex-col overflow-hidden rounded-md border"
      >
        {accordionList.map((item) => (
          <Accordion.Item key={item.id} index={item.id}>
            {({ isOpen }) => (
              <React.Fragment>
                <Accordion.Trigger>
                  <div
                    className={`${isOpen && 'text-blue-600'} cursor-pointer border-y px-4 py-3 text-left`}
                  >
                    {item.title}
                  </div>
                </Accordion.Trigger>

                <Accordion.Content>
                  <motion.div
                    className="w-full break-words bg-gray-50 px-4 py-3"
                    {...animationProps}
                  >
                    {item.content}
                  </motion.div>{' '}
                </Accordion.Content>
              </React.Fragment>
            )}
          </Accordion.Item>
        ))}
      </Accordion>
    </div>
  )
}
```
