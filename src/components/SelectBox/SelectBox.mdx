# SelectBox

사용자가 리스트에서 옵션을 선택할 수 있도록 목록을 표시해주는 인터페이스 요소

## Example

```jsx
import { SelectBox } from 'ventileco-ui'

const selectBoxList = [
  { value: 'apple', label: 'Apple', disabled: true },
  { value: 'kiwi', label: 'Kiwi' },
  { value: 'peach', label: 'Peach', disabled: true },
  { value: 'grape', label: 'Grape' },
  { value: 'aloe', label: 'Aloe' },
  { value: 'apple2', label: 'Apple2' },
  { value: 'banana2', label: 'Banana2', disabled: true },
  { value: 'kiwi2', label: 'Kiwi2' },
  { value: 'peach2', label: 'Peach2', disabled: true },
  { value: 'grape2', label: 'Grape2' },
  { value: 'aloe2', label: 'Aloe2' },
]

function Example() {
  const [value, setValue] = useState<string>()
  const selectRef = useRef<HTMLButtonElement>(null)

  const animationProps = {
    variants: {
      hidden: { opacity: 0, y: 20 },
      visible: { opacity: 1, y: 0 },
    },
    initial: 'hidden',
    animate: 'visible',
    viewport: { once: true },
    transition: { duration: 0.3 },
  }

  return (
    <SelectBox.Input
        placeholder="🐝 Fruits"
        className="flex h-12 w-full items-center rounded-md border border-black px-3"
      />

      <SelectBox.List
        as={motion.ul}
        {...animationProps}
        className='bg-white" absolute z-[200] max-h-[240px] w-full overflow-hidden overflow-y-scroll rounded-md border'
      >
        {({ optionList }) => (
          <>
            {optionList.map((item) => (
              <SelectBox.Item key={item.value} item={item}>
                {({ isSelected, isDisabled, isFocused }) => (
                  <div
                    className={`${isSelected && 'text-blue-600'} ${isDisabled && 'text-gray-300'} ${isFocused && 'bg-blue-100'} w-full px-3 py-2 hover:bg-gray-100`}
                  >
                    {item.label}
                  </div>
                )}
              </SelectBox.Item>
            ))}
          </>
        )}
      </SelectBox.List>
    </SelectBox>
  )
}
```
