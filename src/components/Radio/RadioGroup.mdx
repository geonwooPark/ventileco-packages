# RadioGroup

## With React Hook Form Example

```jsx
import RadioGroup from 'ventileco-ui'
import { useForm } from 'react-hook-form'

function Example() {
  const { register, handleSubmit } = useForm({
    defaultValues: { number: 'number1' },
  })

  const onSubmit = (data: any) => console.log(data)

  const radioRegister = register('number', {
    required: '적어도 한 개를 선택해주세요.',
  })

  return (
    <form onSubmit={handleSubmit(onSubmit)} defaultValue='number1'>
      <RadioGroup register={radioRegister}>
        <RadioGroup.Title>Radio</RadioGroup.Title>
        <RadioGroup.List className="flex gap-4">
          <RadioGroup.Item id="0" value={'number1'}>
```
