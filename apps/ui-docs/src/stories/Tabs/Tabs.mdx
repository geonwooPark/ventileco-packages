# Tabs

클릭 등의 상호작용을 감지하여 특정 섹션을 보여주는 동작을 수행하는 인터페이스 요소

## Example

```jsx
import { Tabs } from 'ventileco-ui'

const tabList = [
  { id: 0, title: '🍇 Grape' },
  { id: 1, title: '🍎 Apple' },
  { id: 2, title: '🍋 Lemon' },
  { id: 3, title: '🍒 Cherry' },
  { id: 4, title: '🍑 Peach' },
]

export default function Example() {
  const [currentTab, setCurrentTab] = useState(0)

  const onChange = (index: number) => {
    setCurrentTab(index)
  }

  return (
    <Tabs
      currentTab={currentTab}
      onChange={onChange}
      className="w-[400px] border rounded-md overflow-hidden"
    >
      <Tabs.Container>
        <Tabs.List>
          {tabList.map((item) => (
            <Tabs.Item key={item.id}>
              {({ isActive }) => (
                <div
                  className={`${isActive && 'text-blue-600'} cursor-pointer px-4 py-3 text-center transition duration-200 hover:opacity-50`}
                >
                  {item.title}
                </div>
              )}
            </Tabs.Item>
          ))}
        </Tabs.List>
        <Tabs.Indicator className="h-[2px] bg-black" />
      </Tabs.Container>

      <Tabs.Content>
        <Tabs.Panel className="px-4 py-3">
          <div>
            Content1Content1ContContent1Content1ContContent1Content1ContContent1Content1Cont
          </div>
        </Tabs.Panel>
        <Tabs.Panel className="px-4 py-3">
          <div>Content2</div>
        </Tabs.Panel>
        <Tabs.Panel className="px-4 py-3">
          <div>Content3</div>
        </Tabs.Panel>
        <Tabs.Panel className="px-4 py-3">
          <div>Content4</div>
        </Tabs.Panel>
        <Tabs.Panel className="px-4 py-3">
          <div>Content5</div>
        </Tabs.Panel>
      </Tabs.Content>
    </Tabs>
  )
}
```
