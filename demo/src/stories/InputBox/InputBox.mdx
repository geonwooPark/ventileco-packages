# InputBox

사용자의 클릭 이벤트를 처리하여 다양한 동작을 수행하는 인터페이스 요소

## Example

```jsx
import InputBox from 'ventileco-ui'

function Example() {
  const [value, setValue] = useState('')

  const onChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { value } = e.target
    setValue(value)
  }

  return (
    <InputBox
      value={value}
      onChange={onChange}
      placeholder="내용을 입력하세요."
      className="h-12 w-[360px] rounded-md border border-black px-3 focus-within:border-2"
      inputClassName="placeholder:text-gray-300 placeholder:text-sm"
      startIcon={...insert your SVG icon...}
    />
  )
}
```
